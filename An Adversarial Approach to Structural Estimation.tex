\subsection{3.1 Logistic Location Model}

\begin{frame}{3.1 Logistic Location Model (Setup)}
  \begin{itemize}
    \item Data: $X_i \sim \text{Logistic}(0,1)$.
    \item Oracle discriminator: 
    $$D_\theta(x)=\frac{p_0(x)}{p_0(x)+p_\theta(x)}=\Lambda\left(-\theta-2\log(1+e^{-x})+2\log(1+e^{-(x-\theta)})\right)$$
    where $\Lambda(x)=1/(1+e^{-x})$
  \end{itemize}
\end{frame}

\begin{frame}{3.1.1 Correct Specification}
  \begin{itemize}
    \item Structural model: $X_i \sim \text{Logistic}(\theta,1)$ (location shift, variance known).
    \item Simulation: $X_i^\theta = \theta + Z_i$ with $Z_i \sim \text{Logistic}(0,1)$.
    \item $n=m=300$
    \item Logistic discriminator is correctly specified in the sense that, with the right $\lambda$, it is the oracle discriminator.
    \begin{figure}[htbp]
      \centering
      \includegraphics[width=0.7\textwidth]{figure 1.png}
    \end{figure}
    when $m=2n$, SE becomes 1.82
  \end{itemize}
\end{frame}

\begin{frame}{3.1.1 Correct Specification (Cont.)}
  \begin{itemize}
    \item Paper also shows how a shallow NN discriminator behaves
    \begin{figure}[htbp]
      \centering
      \includegraphics[width=0.7\textwidth]{figure 2.png}
    \end{figure}
    \begin{itemize}
      \item Curvature matters but level doesn't.
    \end{itemize}
    \item Paper tried bootstrap with both discriminators
    \begin{itemize}
      \item hold fixed the specification of the discriminator
      \item bootstrap SE is close to the asymptotic SD
    \end{itemize}

  \end{itemize}
\end{frame}

\begin{frame}{3.1.2 Misspecification}
  \begin{itemize}

\subsection{3.2 The Roy Model}


\begin{frame}{3.2 The Roy Model: Setup and Observables}
\begin{itemize}
  \item Economic environment
\begin{itemize}
  \item Two sectors $s\in\{1,2\}$ and two periods $t\in\{1,2\}$.
  \item In each period, the agent chooses a sector to maximize current and discounted future expected wages.
\end{itemize}
  \item Wage equations
\begin{align*}
\log w_{i1s} &= \mu_s + \varepsilon_{i1s}, \\
\log w_{i2s} &= \mu_s + \gamma_s \mathbf{1}\{d_{i1}=s\} + \varepsilon_{i2s},
\end{align*}
where $\mu_s$ is the base wage and $\gamma_s$ captures returns to sector-specific experience.
  \item Information and shocks
\begin{itemize}
  \item Shocks $(\varepsilon_{i11},\varepsilon_{i12},\varepsilon_{i21},\varepsilon_{i22})$ are observed by the agent when decisions are made.
  \item They follow a mean-zero multivariate normal distribution with sector and time correlations $(\rho_s,\rho_t)$.
\end{itemize}
\end{itemize}
\end{frame}

\begin{frame}{3.2 The Roy Model: Setup and Observables (Cont.)}
\begin{itemize}
  \item Choice rules
\begin{align*}
d_{i1} &= \arg\max_{s\in\{1,2\}} \; w_{i1s} + \beta \mathbb{E}[w_{i2}\mid d_{i1}=s], \\
d_{i2} &= \arg\max_{s\in\{1,2\}} \; w_{i2s},
\end{align*}
with discount factor $\beta=0.9$
  \item Observed data
\[
X_i = (\log w_{i1}, d_{i1}, \log w_{i2}, d_{i2}),
\]
  \item parameters
\[
\theta = (\mu_1, \mu_2, \gamma_1, \gamma_2, \sigma_1, \sigma_2, \rho_t, \rho_s).
\]
\end{itemize}

\end{frame}
